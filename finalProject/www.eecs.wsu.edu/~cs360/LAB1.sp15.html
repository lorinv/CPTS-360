2. typedef unsigned int u32;

   The following printu(u32 x) function prints an unsigned int (u32)x:

#define BASE 10

char *table = "0123456789ABCDEF";

int rpu(u32 x)
{
  char c;
  if (x){
     c = table[x % BASE];
     rpu(x / BASE);
     putchar(c);
  }
} 

int printu(u32 x)
{
  if (x==0)
     putchar('0');
  else
     rpu(x);
  putchar(' ');
}
